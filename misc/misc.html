An exercise in utilizing string indices. This was one of the first scripts I practiced with in Python: the concept that a declared string can be treated just like an array can. Each character in the string is assigned an array index (starting at index 0). So I simply created a single string of the alphabet. It asks for a number, then reports the letter that corresponds to the entry. 
# The alphabet string; The _ takes up index 0, so 1 corresponds to 'a,' while 26 corresponds to 'z'
abc = "_abcdefghijklmnopqrstuvwxyz"

# Ask the user to input a number...
entry = int(input("ENTER A NUMBER BETWEEN 1 and 26, AND WE'LL SHOW YOU WHICH LETTER IT CORRESPONDS WITH: "))

# Outputs the correct number
print("The Letter in position #", entry, " is '", abc[entry].upper(), "'")

Experts release sports-betting lines for games and give 1) a projected accumulated total of points between the teams (the "over/under"), and 2) a projected margin of victory relative to a team (+ means they are projected to lose; minus indicates projected victory). So this basic script takes in entry of the home and away teams, the over/under, and the victory margin (w/respect to the home team). It then calculates/outputs the approximate final score and indicates the projected winner. It is normal to get x.5s for answers. This is exactly due to it cannot be the final score, which would prevent ties. 
# "What's the Over and Under?"

away = str(input("Enter the Away Team's name: "))   # Asks for Away Team Name
home = str(input("Enter the Home Team's name: "))   # Asks for the Home Team Name

line = float(input("What is the current over/under 'Vegas' line?: "))   # Input the total points number
margin = float(input("What is the margin (w/respect to the home team)?: "))     # This is the +/- number

win = line / 2 + abs(margin) / 2
lose = line / 2 - abs(margin) / 2

if margin > 0:  #  Using the home team as reference.
    team_win = away # If the home team is NOT favored, they should have a + number beside them
else:
    team_win = home # If they are favored, they'll have a minus number beside them

print("The final score is projected to be {} - {}, in favor of {}".format(win,lose,team_win))



This programming gives you a math quiz. I included choices for the quantity of questions you want on your quiz. It then gives you random +/-/*/div problems. At the end, it gives a grade. I programmed it so you shouldn't need a calculator. 
# MATH QUIZ
# The program selects random integers and quizzes the user
import random   # Imports random function

def howmany():
    qty = int(input("How many questions would you like on the quiz?: "))
    return qty

def addition(p_num):
    r1, r2 = float(random.randint(1, 1000)) / 10.0, float(random.randint(1, 1000)) / 10.0
    ans = round(r1 + r2,1)
    user_ans = float(input("{}. What is {} + {} ?: ".format(p_num,r1,r2)))
    if user_ans == ans:
        print("CORRECT! \n")
        return True
    else:
        print("WRONG! The right answer was: ", round(r1 + r2,1), ".\n")
        return False

def subtraction(p_num):
    r1, r2 = random.randint(0, 1000), random.randint(0, 1000)
    ans = r1 - r2
    user_ans = int(input("{}. What is {} - {} ?: ".format(p_num,r1,r2)))
    if user_ans == ans:
        print("CORRECT! \n")
        return True
    else:
        print("WRONG! The right answer was: ", r1 - r2, ".\n")
        return False

def multiplication(p_num):
    r1, r2 = random.randint(2, 30),random.randint(2, 30)
    ans = r1 * r2
    user_ans = int(input("{}. What is {} * {} ?: ".format(p_num,r1,r2)))
    if user_ans == ans:
        print("CORRECT! \n")
        return True
    else:
        print("WRONG! The right answer was: ", r1 * r2, ".\n")
        return False

def division(p_num):
    r1, r2 = 3,2  # Just for initialization
    while r1 % r2 != 0:     # This will loop until two numbers are selected with a remainder of zero
        r1 = random.randint(1, 100)
        r2 = random.randint(2, 100)
        if r1 == r2:    # This statement basically resets the loop if r1 is equal to r2
            r1 = 3
            r2 = 2            
    ans = r1 / r2
    user_ans = float(input("{}. What is {} / {} ?: ".format(p_num,r1,r2)))
    if user_ans == ans:
        print("CORRECT! \n")
        return True
    else:
        print("WRONG! The right answer was: ", r1 / r2, ".\n")
        return False

# Problem Types: 1 - Addition // 2 - Subtraction // 3 - Multiplication // 4 - Division
p_qty = howmany()   # Will ask and store how many questions will be on the quiz
right = 0           # Represents the counter variable for the questions the user got correct
choice = 'Y'        # Initial value that allows while loop to run

while choice == 'y' or choice == 'Y':
    for new_problem in range(0,p_qty):
        p_type = random.randint(1,4)    # SELECTS RANDOM NUMBER WHICH WILL DICTATE WHAT KIND OF PROBLEM IS PRESENTED
        p = new_problem + 1
        if p_type == 1:
            correct = addition(p)
            if correct == True:
                right += 1
        elif p_type == 2:
            correct = subtraction(p)
            if correct == True:
                right += 1
        elif p_type == 3:
            correct = multiplication(p)
            if correct == True:
                right += 1
        else:
            correct = division(p)
            if correct == True:
                right += 1
    final_score = right / p_qty
    print("Your Final Score was: ",round(final_score * 100,1)," !")
    choice = input("Do you want to take another quiz? (Y/N):")
    if choice == 'y' or choice == 'Y':
        p_num = 1
        right = 0

print("\nTHANK YOU FOR TAKING THE MATH QUIZ !!!")
